JavaScript for循环

for循环注意事项

  • 基本语法
    • for (初始化; 条件; 迭代) { 代码块 }
    • 三个部分都可以省略(分号必须保留)
  • 循环控制
    • break - 立即终止循环
    • continue - 跳过当前迭代
  • 最佳实践
    • 避免无限循环
    • 循环变量通常使用i,j,k等命名
    • 数组遍历优先使用forEach/map等高级方法
  • 性能考虑
    • 缓存数组长度避免重复计算
    • 减少循环内部的计算量

for循环示例

// 1. 基本for循环
for (let i = 0; i < 5; i++) {
    console.log(i); // 0,1,2,3,4
}

// 2. 遍历数组
const fruits = ['苹果', '香蕉', '橙子'];
for (let i = 0; i < fruits.length; i++) {
    console.log(fruits[i]); // 苹果,香蕉,橙子
}

// 3. 嵌套循环
for (let i = 1; i <= 3; i++) {
    for (let j = 1; j <= 3; j++) {
        console.log(`i=${i}, j=${j}`);
    }
}

// 4. break和continue
for (let i = 0; i < 10; i++) {
    if (i === 5) break; // 当i=5时终止循环
    if (i % 2 === 0) continue; // 跳过偶数
    console.log(i); // 1,3
}
