JavaScript while循环

while循环注意事项

  • 基本语法
    • while (条件) { 代码块 }
    • do { 代码块 } while (条件)
  • 与for循环区别
    • 更适合不确定循环次数的场景
    • 需要手动控制循环变量
  • 循环控制
    • break - 立即终止循环
    • continue - 跳过当前迭代
  • 最佳实践
    • 确保循环条件最终会变为false
    • 避免无限循环
    • 复杂条件可提取为变量提高可读性

while循环示例

// 1. 基本while循环
let count = 0;
while (count < 5) {
    console.log(count); // 0,1,2,3,4
    count++;
}

// 2. do-while循环
let num = 5;
do {
    console.log(num); // 5 (至少执行一次)
    num++;
} while (num < 5);

// 3. 遍历链表结构
const list = {
    value: 1,
    next: {
        value: 2,
        next: {
            value: 3,
            next: null
        }
    }
};
let current = list;
while (current) {
    console.log(current.value); // 1,2,3
    current = current.next;
}

// 4. break和continue
let i = 0;
while (i < 10) {
    if (i === 5) break; // 当i=5时终止循环
    if (i % 2 === 0) { // 跳过偶数
        i++;
        continue;
    }
    console.log(i); // 1,3
    i++;
}
