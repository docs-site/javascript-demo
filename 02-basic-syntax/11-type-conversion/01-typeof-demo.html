JavaScript 数据类型检测

JavaScript 数据类型

  • 6种基本数据类型:
    • string
    • number
    • boolean
    • object
    • function
    • symbol
  • 3种对象类型:
    • Object
    • Date
    • Array
  • 2种无值类型:
    • null
    • undefined

类型检测示例

// 1. 基本数据类型检测
const str = "Hello";
const num = 123;
const bool = true;
const sym = Symbol();
const func = function() {};
const obj = {};

// 2. 对象类型检测
const arr = [1, 2, 3];
const date = new Date();

// 3. 无值类型检测
const empty = null;
const notDefined = undefined;

// 检测函数
function getType(value) {
  if (value === null) return "null";
  if (Array.isArray(value)) return "array";
  if (value instanceof Date) return "date";
  return typeof value;
}
