JavaScript 参数(Parameters)与实参(Arguments)详解

基本概念

形式参数(Parameters)：函数定义时声明的变量

实际参数(Arguments)：函数调用时传入的具体值

arguments对象：函数内部可用的类数组对象，包含所有传入参数

基础示例

1. 基本参数传递

// 形式参数 a, b
function sum(a, b) {
    return a + b;
}

// 实际参数 3, 5
const result = sum(3, 5);
点击按钮查看结果

2. 参数数量不匹配

function showParams(a, b) {
    console.log("参数a:", a);
    console.log("参数b:", b);
    console.log("arguments:", arguments);
}

showParams(1);       // 少传参数
showParams(1, 2, 3); // 多传参数
点击按钮查看结果

高级特性

3. arguments对象

function logArguments() {
    console.log("参数个数:", arguments.length);
    console.log("所有参数:", arguments);
    
    // 遍历arguments
    let sum = 0;
    for (let i = 0; i < arguments.length; i++) {
        sum += arguments[i];
    }
    return sum;
}
点击按钮查看结果

4. 默认参数

// ES5实现默认参数
function greet(name) {
    name = name || 'Guest';
    return `Hello, ${name}!`;
}

// ES6默认参数
function greetES6(name = 'Guest') {
    return `Hello, ${name}!`;
}

剩余参数与展开语法

5. 剩余参数(Rest Parameters)

function sumAll(...numbers) {
    return numbers.reduce((acc, curr) => acc + curr, 0);
}

// 与arguments对比
function sumArguments() {
    return Array.from(arguments).reduce((acc, curr) => acc + curr, 0);
}
点击按钮查看结果

6. 展开语法(Spread Syntax)

const numbers = [1, 2, 3, 4];

function multiply(a, b, c, d) {
    return a * b * c * d;
}

// 使用展开语法传递数组
const result = multiply(...numbers);
点击按钮查看结果

参数传递方式

特性 形式参数(Parameters) 实参(Arguments) arguments对象
定义位置 函数声明时 函数调用时 函数内部自动创建
数据类型 变量名 具体值 类数组对象
默认值 可设置(ES6) 无 无
严格模式 不受影响 不受影响 不与参数绑定
箭头函数 支持 支持 不支持
